% Custom commands for message formatting
\newcommand{\sentmessage}[2]{%
    \begin{flushright}
    \colorbox{sentmessage!20}{%
        \parbox{0.7\textwidth}{%
            \textcolor{black}{#1}%
        }%
    }\\[0.1cm]
    {\small\textcolor{timestampgray}{#2}}
    \end{flushright}
    \vspace{0.2cm}
}

    \vspace{0.2cm}
}
